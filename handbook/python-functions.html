1. Defining Functions

Functions let you name and reuse a block of code. The def keyword defines one.

# Basic function
def greet():
    print("Hello, World!")

greet()   # Call it

# With a docstring (ALWAYS write these)
def calculate_bmi(weight_kg, height_m):
    """Calculate Body Mass Index.
    
    Args:
        weight_kg: Weight in kilograms.
        height_m: Height in meters.
    
    Returns:
        BMI as a float, rounded to 1 decimal place.
    """
    bmi = weight_kg / (height_m ** 2)
    return round(bmi, 1)

result = calculate_bmi(70, 1.75)
print(result)   # 22.9
help(calculate_bmi)   # Shows the docstring

2. Arguments (The Full Picture)

Positional Arguments

def add(a, b):
    return a + b

add(3, 5)      # a=3, b=5  → 8
add(5, 3)      # a=5, b=3  → 8 (same result here)
# Position matters for non-commutative operations:
def divide(a, b):
    return a / b
divide(10, 2)  # 5.0
divide(2, 10)  # 0.2  — different!

Keyword Arguments

def create_user(name, age, role):
    return {"name": name, "age": age, "role": role}

# Keyword args can be in any order
create_user(role="admin", name="Andi", age=25)

# Mix positional and keyword (positional must come first)
create_user("Andi", age=25, role="admin")

Default Arguments

def greet(name, greeting="Hello", punctuation="!"):
    return f"{greeting}, {name}{punctuation}"

greet("Andi")                     # "Hello, Andi!"
greet("Andi", "Hi")               # "Hi, Andi!"
greet("Andi", punctuation=".")   # "Hello, Andi."
❌ THE MUTABLE DEFAULT ARGUMENT TRAP — Read This Carefully 
# WRONG — the list is created ONCE and shared across all calls!
def add_item(item, items=[]):
    items.append(item)
    return items

print(add_item("a"))   # ['a']
print(add_item("b"))   # ['a', 'b']  ← surprise!
print(add_item("c"))   # ['a', 'b', 'c']  ← it accumulates!

# CORRECT — use None as sentinel, create fresh list inside
def add_item(item, items=None):
    if items is None:
        items = []
    items.append(item)
    return items

*args — Variable Positional Arguments

# *args collects extra positional args into a TUPLE
def sum_all(*args):
    print(type(args))    # <class 'tuple'>
    return sum(args)

sum_all(1, 2, 3)          # 6
sum_all(1, 2, 3, 4, 5)    # 15

# Combining with regular args
def log(level, *messages):
    for msg in messages:
        print(f"[{level}] {msg}")

log("INFO", "Server started", "Listening on port 8080")

# Unpacking a list into positional args
nums = [1, 2, 3]
sum_all(*nums)   # Same as sum_all(1, 2, 3)

**kwargs — Variable Keyword Arguments

# **kwargs collects extra keyword args into a DICT
def print_info(**kwargs):
    print(type(kwargs))    # <class 'dict'>
    for key, val in kwargs.items():
        print(f"  {key}: {val}")

print_info(name="Andi", age=25, city="Jakarta")
# name: Andi
# age: 25
# city: Jakarta

# Unpacking a dict into keyword args
data = {"name": "Andi", "age": 25}
create_user(**data, role="admin")

Full argument order

# Order: positional, *args, keyword-only, **kwargs
def full_example(a, b, *args, keyword_only, **kwargs):
    print(a, b, args, keyword_only, kwargs)

full_example(1, 2, 3, 4, keyword_only="kw", x=10, y=20)
# 1 2 (3, 4) kw {'x': 10, 'y': 20}

# Keyword-only args: after * separator
def connect(host, port, *, timeout=30, retries=3):
    # timeout and retries MUST be passed by keyword
    print(f"Connecting to {host}:{port} (timeout={timeout})")

connect("localhost", 8080, timeout=60)   # ✅
connect("localhost", 8080, 60)           # TypeError! timeout must be keyword

3. Return Values

# No return → returns None implicitly
def say_hi():
    print("Hi")

result = say_hi()
print(result)   # None

# Return multiple values (returns a tuple)
def min_max(numbers):
    return min(numbers), max(numbers)

low, high = min_max([3, 1, 9, 2, 7])
print(low, high)    # 1 9

result = min_max([3, 1, 9])
print(result)       # (1, 9) — it's a tuple

# Early return pattern (guard clauses)
def divide(a, b):
    if b == 0:
        return None   # Exit early
    return a / b

4. Scope — The LEGB Rule

When Python looks up a name, it searches in this order: Local → Enclosing → Global → Built-in.

x = "global"   # G — global scope

def outer():
    x = "enclosing"   # E — enclosing scope
    
    def inner():
        x = "local"   # L — local scope
        print(x)       # "local" — finds L first
    
    inner()
    print(x)           # "enclosing"

outer()
print(x)               # "global"

# B — built-ins: len, print, range, etc.
print(len([1,2,3]))   # len is in built-in scope

global and nonlocal keywords

counter = 0

def increment():
    global counter      # Declare we mean the global one
    counter += 1

increment()
increment()
print(counter)   # 2

# nonlocal — for nested functions
def make_counter():
    count = 0
    def increment():
        nonlocal count   # Refers to enclosing 'count'
        count += 1
        return count
    return increment

counter = make_counter()
print(counter())  # 1
print(counter())  # 2
print(counter())  # 3
⚠️ Avoid global variables Global state makes code hard to test and reason about. Pass values as arguments and return results instead.

5. Functions Are First-Class Objects

In Python, functions are objects. You can assign them to variables, pass them as arguments, and return them from other functions.

# Assign to a variable
def square(x):
    return x ** 2

fn = square
print(fn(5))   # 25

# Pass as argument
def apply(func, value):
    return func(value)

print(apply(square, 4))   # 16
print(apply(str, 42))     # "42"

# Return from a function (closure)
def multiplier(n):
    def multiply(x):
        return x * n    # 'n' is captured from outer scope
    return multiply

double = multiplier(2)
triple = multiplier(3)
print(double(5))   # 10
print(triple(5))   # 15

# Store in a list
operations = [str, int, float, square]
for op in operations:
    print(op(5))

Lambda Functions

# lambda: anonymous single-expression function
square = lambda x: x ** 2
add = lambda a, b: a + b

print(square(4))    # 16
print(add(3, 5))   # 8

# Main use: as key functions for sorting
students = [("Alice", 85), ("Bob", 92), ("Charlie", 78)]
students.sort(key=lambda s: s[1])        # Sort by score
students.sort(key=lambda s: s[1], reverse=True)  # Descending

words = ["banana", "apple", "cherry"]
words.sort(key=lambda w: len(w))   # Sort by length

# With map() and filter()
nums = [1, 2, 3, 4, 5]
squared = list(map(lambda x: x**2, nums))     # [1, 4, 9, 16, 25]
evens   = list(filter(lambda x: x%2==0, nums)) # [2, 4]

6. Recursion

A function that calls itself. Every recursive solution needs: (1) a base case to stop, (2) a recursive case that moves toward the base.

# Factorial: n! = n × (n-1)!
def factorial(n):
    if n <= 1:           # Base case
        return 1
    return n * factorial(n - 1)   # Recursive case

factorial(5)  # 120
# factorial(5)
#   → 5 * factorial(4)
#       → 4 * factorial(3)
#           → 3 * factorial(2)
#               → 2 * factorial(1)
#                   → 1  (base case!)

# Fibonacci
def fib(n):
    if n <= 1:
        return n
    return fib(n-1) + fib(n-2)

# This is slow (exponential) — use memoization
from functools import lru_cache

@lru_cache(maxsize=None)
def fib_fast(n):
    if n <= 1:
        return n
    return fib_fast(n-1) + fib_fast(n-2)

fib_fast(50)   # Instant; without cache would take minutes

# Recursive directory traversal (real-world use)
import os
def list_files(path, indent=0):
    for item in os.listdir(path):
        full = os.path.join(path, item)
        print("  " * indent + item)
        if os.path.isdir(full):
            list_files(full, indent + 1)   # Recurse into subdir
💡 Python's recursion limit Python limits recursion depth to 1000 by default (to prevent stack overflow). For deep recursion, either use iteration or increase the limit with sys.setrecursionlimit(). For heavy use, prefer iterative solutions.

7. Common Mistakes

❌ Mutable default argument (already shown — worth repeating) Never use [], {}, or set() as default arguments. Use None and create inside.
❌ Missing return — function returns None silently 
def double(x):
    x * 2   # ❌ Missing return! Returns None

result = double(5)
print(result)      # None — not 10!
❌ Shadowing built-ins 
# Don't name variables after built-ins
list = [1, 2, 3]      # ❌ Shadows built-in list()
print(list("abc"))    # TypeError! list is now your list, not the built-in

# Also avoid: input, print, id, type, str, int, sum, min, max
