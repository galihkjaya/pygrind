1. Creating Strings

# Single or double quotes (equivalent)
s1 = 'Hello'
s2 = "World"

# Use one to avoid escaping the other
s3 = "It's a great day"
s4 = 'He said "hello"'

# Triple quotes — multi-line strings
essay = """This is a
multi-line string.
Preserves whitespace and newlines."""

# Raw strings — backslash is literal
path = r"C:\Users\andi\documents"   # No escape needed
regex = r"\d+\.\d+"                  # For regex patterns

# Byte strings
b = b"hello"   # bytes object, not str

# String repetition and concatenation
"ha" * 3           # "hahaha"
"Hello" + " World"  # "Hello World"

# len()
len("hello")   # 5
len("")       # 0

2. Indexing & Slicing

Strings are sequences. You can access individual characters or ranges with indexing and slicing.

s = "Python"
#    P  y  t  h  o  n
# +  0  1  2  3  4  5
# - -6 -5 -4 -3 -2 -1

# Indexing
s[0]    # 'P'
s[2]    # 't'
s[-1]   # 'n' — last character
s[-2]   # 'o' — second to last

# Slicing: s[start:stop:step]  (stop is exclusive)
s[0:3]   # 'Pyt'
s[2:5]   # 'tho'
s[2:]    # 'thon'  — from index 2 to end
s[:3]    # 'Pyt'   — from start to index 3
s[:]     # 'Python' — full copy
s[::2]   # 'Pto'   — every 2nd character
s[::-1]  # 'nohtyP' — reversed!

# Slicing never raises IndexError
s[100:200]   # '' — empty, not an error
💡 Reverse a string easily s[::-1] is the idiomatic way to reverse a string in Python. It reads "step through with step -1", starting from end to beginning.

3. Essential String Methods

Strings have lots of built-in methods. None of them modify the original — they always return a new string (strings are immutable).

Case Methods

s = "hello World"

s.upper()        # "HELLO WORLD"
s.lower()        # "hello world"
s.title()        # "Hello World"
s.capitalize()   # "Hello world" (only first letter)
s.swapcase()     # "HELLO wORLD"

Search & Check

s = "The quick brown fox"

# Find position
s.find("quick")      # 4 — index of first match
s.find("cat")        # -1 — not found
s.index("quick")     # 4 — like find() but raises ValueError if not found
s.rfind("o")         # 17 — last occurrence
s.count("o")         # 2 — count occurrences

# Check content
s.startswith("The")   # True
s.endswith("fox")     # True
"quick" in s          # True — membership test

# Check character types
"hello".isalpha()    # True — all letters
"123".isdigit()      # True — all digits
"abc123".isalnum()   # True — letters and digits
"   ".isspace()      # True — all whitespace
"HELLO".isupper()    # True
"hello".islower()    # True

Modify / Clean

# Strip whitespace
"  hello  ".strip()    # "hello"
"  hello  ".lstrip()   # "hello  "
"  hello  ".rstrip()   # "  hello"
"...hello...".strip(".")  # "hello" — strip specific chars

# Replace
"hello world".replace("world", "Python")   # "hello Python"
"aabbcc".replace("a", "x", 1)              # "xabbcc" — max 1 replacement

# Split and join
"a,b,c".split(",")      # ['a', 'b', 'c']
"hello world".split()   # ['hello', 'world'] — splits on whitespace
"a\nb\nc".splitlines()  # ['a', 'b', 'c']

# join: the reverse of split
",".join(["a", "b", "c"])    # "a,b,c"
" ".join(["hello", "world"]) # "hello world"
"".join(["a", "b", "c"])     # "abc"

# Padding / alignment
"42".zfill(5)         # "00042"
"hi".ljust(10)        # "hi        "
"hi".rjust(10)        # "        hi"
"hi".center(10)       # "    hi    "
"hi".ljust(10, "*")   # "hi********"

4. String Formatting

There are 3 ways to format strings in Python. Learn all three — you'll see them all in the wild.

f-strings (Python 3.6+) — preferred

name = "Andi"
score = 98.567
items = ["apple", "banana"]

print(f"Hello, {name}!")
print(f"Score: {score:.2f}")         # 98.57 (2 decimal places)
print(f"Items: {len(items)}")        # Expressions work!
print(f"Upper: {name.upper()}")      # Method calls work!
print(f"{1_000_000:,}")             # 1,000,000
print(f"{0.25:.1%}")                # 25.0%
print(f"{'left':<<10}")             # "left      " (left-align, width 10)
print(f"{'right':>10}")             # "     right"
print(f"{score=}")                   # score=98.567 (debug mode)

.format() — older but common in codebases

"{} is {} years old".format("Andi", 25)
# "Andi is 25 years old"

"{name} is {age}".format(name="Andi", age=25)
# "Andi is 25"

"{0:.2f}".format(3.14159)
# "3.14"

% formatting — legacy, still in logging

"Hello, %s! You are %d years old" % ("Andi", 25)
"Pi is %.2f" % 3.14159

5. Strings Are Immutable

s = "hello"
s[0] = "H"   # TypeError: 'str' object does not support item assignment

# Every string "modification" creates a NEW string
s = s.replace("h", "H")   # s is now a new string "Hello"
s = s.upper()               # another new string "HELLO"

# Building strings in a loop — use a list and join()
# Bad — creates many intermediate strings
result = ""
for i in range(1000):
    result += str(i)   # ❌ O(n²) memory

# Good — collect in list, join once
parts = []
for i in range(1000):
    parts.append(str(i))
result = "".join(parts)   # ✅ O(n)

6. Encoding & Unicode

# Python 3 strings are Unicode by default
s = "こんにちは"   # Works fine
len(s)            # 5 — character count, not bytes

# Convert between str and bytes
s = "hello"
b = s.encode("utf-8")    # b'hello' — bytes
s2 = b.decode("utf-8")   # "hello" — back to str

# ord() and chr()
ord("A")    # 65  — Unicode code point
chr(65)    # 'A' — back to character
ord("a")    # 97

# Escape sequences
"\n"     # newline
"\t"     # tab
"\\"     # literal backslash
"\'"     # single quote
"\""     # double quote
"\u0041" # Unicode: 'A'
